    try:
        refresh_all_projects(db, github_token=settings.github_token)
    finally:
        db.close()


scheduler = BackgroundScheduler()


@asynccontextmanager
async def lifespan(app: FastAPI):
    scheduler.add_job(
        _scheduled_refresh,
        "interval",
        minutes=settings.refresh_interval_minutes,
        id="interval_refresh",

    yield
    scheduler.shutdown()

    description="API for the RHDP Publishing House content lifecycle portal",

```python
"""RCARS MCP tools for content advisory queries.

Each tool instantiates its own RCARSClient since MCP tools run
outside the FastAPI request lifecycle. Tools catch RCARSError
and return structured error dicts for graceful degradation.
"""
from __future__ import annotations

from app.core.config import settings
from app.mcp.server import mcp
from app.services.rcars_client import RCARSClient, RCARSError


@mcp.tool()
async def ph_rcars_query(query: str) -> dict:
    """Submit a content vetting query to RCARS and return matching catalog items
    with relevance tiers and rationale.

    Use this tool during intake vetting to check for existing content that
    overlaps with a proposed workshop or demo. The tool handles advisor job
    submission, polling, and result formatting internally.

    Args:
        query: Natural language description combining learning objectives,
               topics, and products for the content being vetted.
    """
    client = RCARSClient(base_url=settings.rcars_internal_url)
    try:
        result = await client.query_advisor(query)
        return result
    except RCARSError as e:
        return {"error": str(e), "status": "unavailable"}


@mcp.tool()
async def ph_rcars_catalog_search(query: str = "", limit: int = 20) -> dict:
    """Search the RCARS catalog for existing content items.

    Returns paginated catalog items with basic metadata. Use this to browse
    available workshops, demos, and labs in the Red Hat Demo Platform catalog.

    Args:
        query: Optional search text to filter catalog items.
        limit: Maximum number of items to return (default: 20).
    """
    client = RCARSClient(base_url=settings.rcars_internal_url)
    try:
        result = await client.catalog_search(query=query or None, limit=limit)
        return result
    except RCARSError as e:
        return {"error": str(e), "status": "unavailable"}


@mcp.tool()
async def ph_rcars_catalog_item(ci_name: str) -> dict:
    """Get full metadata and analysis for a specific RCARS catalog item.

    Returns the complete catalog item including display name, stage,
    analysis summary, content hash, staleness info, and tags.

    Args:
        ci_name: The unique catalog item name (e.g., 'ocp4_getting_started').
    """
    client = RCARSClient(base_url=settings.rcars_internal_url)
    try:
        result = await client.catalog_item(ci_name)
        return result
    except RCARSError as e:
        return {"error": str(e), "status": "unavailable"}
```

    status: str  # "ok" or "unavailable"

class HealthResponse(BaseModel):

```python
@router.get("/health", response_model=HealthResponse)
async def health_check():
    rcars_status = None
    if settings.rcars_internal_url:
        client = RCARSClient(base_url=settings.rcars_internal_url)
        try:
            result = await client.health()
            rcars_status = RCARSHealthStatus(status=result.get("status", "ok"))
        except Exception as e:
            rcars_status = RCARSHealthStatus(status="unavailable", error=str(e))
    return HealthResponse(status="ok", rcars=rcars_status)
```
